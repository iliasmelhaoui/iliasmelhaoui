%% Début des exercices
\exercice{1}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $A = 3x + 2y + 4x - y + 5x$
\item $B = 6a - 2b + 3a + 4b - a$
\item $C = 5p + 3q - 2p + 4q - q$
\item $D = 7k - 5 + 2k + 3 - 4k$
\end{enumerate}
\end{multicols}

\exercice{2}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $E = 4x + 3 + 2x - 5 + x$
\item $F = 8y - 2y + 3 - 4y + 7$
\item $G = 5a + 2b - 3a + 4b - b$
\item $H = 6m + 3n - 2m - 4n + 5m$
\end{enumerate}
\end{multicols}

\exercice{3}
Problème :
\begin{enumerate}[label=\alph*)]
\item Un rectangle a pour longueur $L = 3x + 2$ et pour largeur $l = 2x + 1$. Exprime son périmètre $P$ en fonction de $x$ et réduis l'expression.
\item Un carré a pour côté $c = 4y - 3$. Exprime son périmètre $P$ en fonction de $y$ et réduis l'expression.
\end{enumerate}

\exercice{4}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $I = 2a + 3b + 4a - 5b + 6a$
\item $J = 7x - 3y + 2x + 4y - 5x$
\item $K = 3p + 2q - p + 4q - 3q$
\item $L = 5k + 3m - 2k + 4m - 6k$
\end{enumerate}
\end{multicols}

\exercice{5}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $M = 4x + 3 + 2x - 5 + x - 2$
\item $N = 8y - 2y + 3 - 4y + 7 - 1$
\item $O = 5a + 2b - 3a + 4b - b + 6$
\item $P = 6m + 3n - 2m - 4n + 5m - 1$
\end{enumerate}
\end{multicols}
