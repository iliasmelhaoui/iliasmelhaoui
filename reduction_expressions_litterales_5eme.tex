\exercice{1}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $A = 3x + 2y + 4x - y + 5x$
\item $B = 6a - 2b + 3a + 4b - a$
\item $C = 5p + 3q - 2p + 4q - q$
\item $D = 7k - 5 + 2k + 3 - 4k$
\end{enumerate}
\end{multicols}

\exercice{2}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $E = 4x + 3 + 2x - 5 + x$
\item $F = 8y - 2y + 3 - 4y + 7$
\item $G = 5a + 2b - 3a + 4b - b$
\item $H = 6m + 3n - 2m - 4n + 5m$
\end{enumerate}
\end{multicols}

\exercice{3}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $I = 2a + 3b + 4a - 5b + 6a$
\item $J = 7x - 3y + 2x + 4y - 5x$
\item $K = 3p + 2q - p + 4q - 3q$
\item $L = 5k + 3m - 2k + 4m - 6k$
\end{enumerate}
\end{multicols}

\exercice{4}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $M = 4x + 3 + 2x - 5 + x - 2$
\item $N = 8y - 2y + 3 - 4y + 7 - 1$
\item $O = 5a + 2b - 3a + 4b - b + 6$
\item $P = 6m + 3n - 2m - 4n + 5m - 1$
\end{enumerate}
\end{multicols}

\exercice{5}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $Q = 3x^2 + 2x^2 - x^2 + 5x$
\item $R = 4x^3 - 2x^3 + 3x^2 + x$
\item $S = 2x^2 + 5x + 3x^2 - 2x$
\item $T = 6x^3 - 4x^3 + 2x^2 + 3x$
\end{enumerate}
\end{multicols}

\exercice{6}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $U = 2a \times 2a + 3b \times b$
\item $V = 5x \times 3x - 2y \times y$
\item $W = 4a \times a + 2b \times 3b$
\item $X = 3x \times 2x - 4y \times y + x \times x$
\end{enumerate}
\end{multicols}

\exercice{7}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $Y = 2x^2 + 3x \times x - 4x + 5$
\item $Z = 5x^3 - 2x \times x^2 + 3x^2$
\item $A_1 = 4a \times a + 2a^2 - 3a + 1$
\item $B_1 = 6x^2 - 3x \times x + 2x - 4$
\end{enumerate}
\end{multicols}

\exercice{8}
Problème :
\begin{enumerate}[label=\alph*)]
\item Un segment est composé de trois parties : $2x + 3$, $x + 5$ et $3x - 2$. Exprime la longueur totale du segment en fonction de $x$ et réduis l'expression.
\end{enumerate}

\exercice{9}
Problèmes :
\begin{enumerate}[label=\alph*)]
\item Un chemin est formé de deux tronçons : $5x + 7$ et $3x - 4$. Exprime la longueur totale du chemin en fonction de $x$ et réduis l'expression.
\item Une ligne brisée est composée de trois segments : $4x + 1$, $2x + 6$ et $x - 3$. Exprime la longueur totale de la ligne brisée en fonction de $x$ et réduis l'expression.
\end{enumerate}

\exercice{10}
Problèmes originaux :
\begin{enumerate}[label=\alph*)]
\item Paul a $3x + 5$ billes et Marie a $2x + 8$ billes. Exprime le nombre total de billes en fonction de $x$ et réduis l'expression.
\item Un train parcourt $4x + 12$ km le matin et $3x - 5$ km l'après-midi. Exprime la distance totale parcourue en fonction de $x$ et réduis l'expression.
\item Un rectangle a un périmètre de $10x + 20$. Sachant que sa longueur est $4x + 5$, exprime sa largeur en fonction de $x$ et réduis l'expression.
\end{enumerate}
