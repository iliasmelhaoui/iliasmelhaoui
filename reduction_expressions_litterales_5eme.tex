\section*{Réduction d'expressions littérales - Niveau 5ème}
\vspace{0.5cm}

\begin{center}
\fbox{\begin{minipage}{0.9\textwidth}
\textbf{Objectif :} Savoir simplifier une expression littérale en regroupant les termes semblables.
\end{minipage}}
\end{center}

\vspace{0.5cm}

%% Consignes générales
\begin{center}
\fbox{\begin{minipage}{0.9\textwidth}
\textbf{Consignes :}
\begin{itemize}
\item Réduis les expressions suivantes en regroupant les termes de même nature.
\item N'oublie pas de copier l'expression initiale avant de la simplifier.
\item Attention aux signes devant les termes !
\end{itemize}
\end{minipage}}
\end{center}

\vspace{0.5cm}

%% Début des exercices

\exercice{1}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $A = 3x + 2x + 5x$
\item $B = 7y - 4y + y$
\item $C = 2a + 3b + 4a + 5b$
\item $D = 6m - 2m + 4n - n$
\end{enumerate}
\end{multicols}

\exercice{2}
Réduis les expressions suivantes en faisant attention aux signes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $E = 5x - 3x + 2x - 4x$
\item $F = 8a + 3b - 5a - 2b$
\item $G = 4p - 2q + 3p + 7q$
\item $H = 9k - 5k - 3k + 2k$
\end{enumerate}
\end{multicols}

\exercice{3}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $I = 2x + 3 + 4x + 5$
\item $J = 7y - 2 + 3y - 4$
\item $K = 5a + 2b + 3a - 4b + 6$
\item $L = 8m - 3n + 2m + 5n - 1$
\end{enumerate}
\end{multicols}

\exercice{4}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $M = 3x + 2y + 4x - y + 5x$
\item $N = 6a - 2b + 3a + 4b - a$
\item $O = 5p + 3q - 2p + 4q - q$
\item $P = 7k - 5 + 2k + 3 - 4k$
\end{enumerate}
\end{multicols}

\exercice{5}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $Q = 4x + 3 + 2x - 5 + x$
\item $R = 8y - 2y + 3 - 4y + 7$
\item $S = 5a + 2b - 3a + 4b - b$
\item $T = 6m + 3n - 2m - 4n + 5m$
\end{enumerate}
\end{multicols}

\exercice{6}
Problème :
\begin{enumerate}[label=\alph*)]
\item Un rectangle a pour longueur $L = 3x + 2$ et pour largeur $l = 2x + 1$. Exprime son périmètre $P$ en fonction de $x$ et réduis l'expression.
\item Un carré a pour côté $c = 4y - 3$. Exprime son périmètre $P$ en fonction de $y$ et réduis l'expression.
\end{enumerate}

\exercice{7}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $U = 2a + 3b + 4a - 5b + 6a$
\item $V = 7x - 3y + 2x + 4y - 5x$
\item $W = 3p + 2q - p + 4q - 3q$
\item $X = 5k + 3m - 2k + 4m - 6k$
\end{enumerate}
\end{multicols}

\exercice{8}
Réduis les expressions suivantes :
\begin{multicols}{2}
\begin{enumerate}[label=\alph*)]
\item $Y = 4x + 3 + 2x - 5 + x - 2$
\item $Z = 8y - 2y + 3 - 4y + 7 - 1$
\item $A_1 = 5a + 2b - 3a + 4b - b + 6$
\item $B_1 = 6m + 3n - 2m - 4n + 5m - 1$
\end{enumerate}
\end{multicols}
